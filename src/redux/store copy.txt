export const store = configureStore({
  reducer: {
    contacts: combineReducers({
      items: itemsReducer,
    }),
    a: (prevA = '1', action) => {
      if (action.type === 'addAction') {
        return prevA + action.payload;
      }
      return prevA;
    },
  },
});
